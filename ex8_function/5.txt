CREATE OR REPLACE FUNCTION sameset(PAR IN VARCHAR2) RETURN VARCHAR2
    IS
  C_NAME varchar2(10);
    M_NAME  varchar2(10);
    C_CITY varchar2(10);
    E_CITY varchar2(10);
    BEGIN
    SELECT CNAME INTO C_NAME FROM WORKS GROUP BY CNAME HAVING COUNT(ENAME)>1;
    SELECT DISTINCT MNAME INTO M_NAME FROM MANAGES WHERE MNAME IN (SELECT ENAME FROM WORKS WHERE CNAME=C_NAME);
   SELECT CITY INTO C_CITY FROM COMPANY WHERE CNAME=C_NAME;
SELECT CITY INTO E_CITY FROM EMPLOYEE WHERE ENAME=M_NAME;
  IF C_CITY = E_CITY THEN
  	UPDATE EMPLOYEE SET CITY = 'mdu' WHERE ENAME=M_NAME;   
   RETURN 'TRUE';
  ELSE
  RETURN 'FALSE';
  END IF;
  END;
  /
   DECLARE
    an varchar2(10);
   begin
    an:=sameset('mdu');
    dbms_output.put_line('answer : '||an);
   end;
